  \begin{center}
    \begin{tikzpicture}[auto, node distance=2.5cm,>=latex']

      \node [bigblock, right of=e] (W) {\only<1>{$W$}\only<2>{$6/10$}\only<3>{$3/10$}\only<4>{$1/10$}};

      \node [left of=e, node distance=1.5cm] (i) {\only<1>{$i$}\only<2-4>{\alert{a}}};
      \node [right of=d,  node distance=1.5cm] (j) {\only<1>{$j$}\only<2-3>{\alert{a}}\only<4>{\alert{b}}};
      \draw [draw,->] (i) -- (e);
      \draw [draw,->] (d) -- (j);
    \end{tikzpicture}

    \bigskip

    Trouver $e$ et $d$ qui maximise la probabilité d'avoir $j=i$.
  \end{center}
  %Envoi de $k$ messages uniformes, trouver meilleur e/d pour maximiser la proba de succès (2 types de caractère: lettres VS nombres)
\end{frame}

\begin{frame}{Résolution \cite{BF18}}
Problème difficile, impossible de trouver efficacement (ie. NP difficile) le meilleur

Par contre, on peut trouver efficacement une $1-e^{-1} \simeq 0.63$ approx, ie. qui trouve une solution qui fait au moins $63\%$ aussi bien que le meilleur, et on ne peut pas faire mieux
\end{frame}
